#### 02 - Orientação a Objetos com ES5

```js
function Pessoa(name, lastName) {
    console.log("Sou o contrutor...");
    this.name = name;
    this.lastName = lastName;

    this.fullName = function () {
        return `${this.name} ${this.lastName}`;
    };

var jose = new Pessoa("José", "Malcher JR")
```
